## Architecture Overview

### Technology Stack

| Component | Technology | Rationale |
|-----------|-----------|-----------|
| **WebSocket Server** | python-socketio + FastAPI | Native async support, auto-scaling with Redis |

#### Task 2: Connection Manager
**File**: `backend/services/websocket_manager.py`
```python
class WebSocketManager:
    def __init__(self):
        self.active_users = {}  # {user_id: {socket_id, username, avatar}}
        self.post_rooms = {}    # {post_id: [user_id, ...]}

    async def connect(self, user_id, socket_id, username):
        if user_id not in self.active_users:
            self.active_users[user_id] = []
        self.active_users[user_id].append({
            'socket_id': socket_id,
            'username': username,
            'connected_at': utcnow()
        })

    async def disconnect(self, user_id, socket_id):
        if user_id in self.active_users:
            self.active_users[user_id] = [
                u for u in self.active_users[user_id]
                if u['socket_id'] != socket_id
            ]

    async def join_post(self, post_id, user_id):
        if post_id not in self.post_rooms:
            self.post_rooms[post_id] = []
        if user_id not in self.post_rooms[post_id]:
            self.post_rooms[post_id].append(user_id)
```

### Phase 3b: Frontend WebSocket Client (Day 2)

**File**: `frontend/utils/websocket.js`
```javascript
import io from 'socket.io-client';

class WebSocketClient {
  constructor(url = process.env.REACT_APP_WEBSOCKET_URL) {
    this.socket = io(url, {
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionDelayMax: 5000,
      reconnectionAttempts: 5,
    });

    this.setupListeners();
  }

  setupListeners() {
    this.socket.on('connect', () => console.log('Connected'));
    this.socket.on('disconnect', () => console.log('Disconnected'));
    this.socket.on('comment_added', (data) => {
      // Trigger UI update
      window.dispatchEvent(
        new CustomEvent('comment_added', { detail: data })
      );
    });
  }

  joinPost(postId) {
    this.socket.emit('join_post', { post_id: postId });
  }

  leavePost(postId) {
    this.socket.emit('leave_post', { post_id: postId });
  }

  addComment(postId, content, isSuggestion = false) {
    this.socket.emit('add_comment', {
      post_id: postId,
      content,
      is_suggestion: isSuggestion
    });
  }
}

export default new WebSocketClient();
```

function OnlineUsers({ postId }) {
  const [users, setUsers] = useState([]);

  useEffect(() => {
    websocket.socket.on('online_users', (data) => {
      if (data.post_id === postId) {
        setUsers(data.users);
      }
    });
  }, [postId]);

  return (
    <div className="flex gap-2">
      {users.map(user => (
        <div key={user.id} className="flex items-center gap-1 bg-green-900 px-2 py-1 rounded">
          <div className="w-2 h-2 bg-green-500 rounded-full"></div>
          <span className="text-sm text-green-100">{user.username}</span>
        </div>
      ))}
    </div>
  );
}
